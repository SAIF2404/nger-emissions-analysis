-- Query 1: Count total corporations
SELECT COUNT(*) AS total_corporations
FROM emissions_2324;

-- Query 2: Top 10 emitters by Scope 1
SELECT "Organisation name",
CAST(REPLACE("Total scope 1 emissions (t CO2-e)", ',', '') AS REAL) AS scope1
FROM emissions_2324
ORDER BY scope1 DESC
LIMIT 10;

-- Query 3: National Scope 1 total
SELECT CAST(SUM(REPLACE("Total scope 1 emissions (t CO2-e)", ',', '')) AS REAL) AS total_scope1
FROM emissions_2324;

-- RIGHT -- comma because Scope 2 follows it
SELECT CAST(SUM(REPLACE("Total scope 1 emissions (t CO2-e)", ',', '')) AS REAL) AS total_scope1,
       CAST(SUM(REPLACE("Total scope 2 emissions (t CO2-e)", ',', '')) AS REAL) AS total_scope2
FROM emissions_2324;
